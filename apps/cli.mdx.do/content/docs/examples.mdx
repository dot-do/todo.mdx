# Examples

Learn by example - see how to build common CLI patterns with cli.mdx.

## File Processor

```mdx
import { Command, Arg, Flag } from 'cli.mdx'

<Command name="process" description="Process a file">
  <Arg name="input" description="Input file path" required />
  <Flag name="--output" alias="-o" description="Output file path" />
  <Flag name="--format" description="Output format" default="json" />
</Command>

export async function run({ input, output, format }) {
  const data = await readFile(input)
  const processed = processData(data, format)

  if (output) {
    await writeFile(output, processed)
    return `Processed ${input} â†’ ${output}`
  }

  return processed
}
```

## Interactive Config

```mdx
import { Command, Select, TextInput } from 'cli.mdx'

<Command name="init" description="Initialize configuration">
  <TextInput name="projectName" label="Project name:" required />
  <Select
    name="framework"
    label="Choose framework:"
    options={[
      { label: 'Next.js', value: 'nextjs' },
      { label: 'Remix', value: 'remix' },
      { label: 'Astro', value: 'astro' }
    ]}
  />
</Command>

export async function run({ projectName, framework }) {
  const config = {
    name: projectName,
    framework,
    createdAt: new Date().toISOString()
  }

  await writeFile('config.json', JSON.stringify(config, null, 2))

  return `Created ${projectName} with ${framework}`
}
```

## Multi-Command CLI

```mdx
import { CLI, Command, Arg } from 'cli.mdx'

<CLI name="mycli" version="1.0.0">
  <Command name="start" description="Start the server">
    <Flag name="--port" alias="-p" type="number" default={3000} />
  </Command>

  <Command name="build" description="Build for production">
    <Flag name="--minify" type="boolean" default={true} />
  </Command>

  <Command name="test" description="Run tests">
    <Arg name="pattern" description="Test file pattern" />
  </Command>
</CLI>

export const commands = {
  async start({ port }) {
    return `Server started on port ${port}`
  },

  async build({ minify }) {
    return `Built with minify: ${minify}`
  },

  async test({ pattern }) {
    return `Running tests: ${pattern || 'all'}`
  }
}
```

## With Progress

```mdx
import { Command, Spinner, ProgressBar } from 'cli.mdx'

<Command name="download" description="Download files">
  <Arg name="url" required />
</Command>

export async function run({ url }, { render }) {
  // Show spinner while fetching metadata
  render(<Spinner label="Fetching metadata..." />)
  const { size } = await fetchMetadata(url)

  // Show progress bar during download
  let downloaded = 0
  for await (const chunk of downloadStream(url)) {
    downloaded += chunk.length
    render(<ProgressBar total={size} current={downloaded} />)
  }

  return 'Download complete!'
}
```
